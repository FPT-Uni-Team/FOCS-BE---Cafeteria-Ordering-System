# Build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy source code
COPY . .

# Build
RUN dotnet build FOCS.sln -c Release -o /app/build

# Publish main API project (adjust if needed)
RUN dotnet publish ./FOCS/FOCS.csproj -c Release -o /app/publish /p:UseAppHost=false

# Runtime stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app/publish .

ENTRYPOINT ["dotnet", "FOCS.dll"]
